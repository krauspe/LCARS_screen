import sys
from PySide6.QtWidgets import (QApplication, QMainWindow, QWidget, QVBoxLayout, 
                             QHBoxLayout, QLabel, QPushButton, QFrame, QGridLayout)
from PySide6.QtCore import Qt

# LCARS Color Palette
COLOR_ORANGE = "#ff9900"
COLOR_MAUVE  = "#cc6699"
COLOR_BLUE   = "#9999ff"
COLOR_PEACH  = "#ffcc99"
COLOR_BG     = "#000000"

class LcarsElbow(QFrame):
    """Custom widget for the signature LCARS 'L' shape."""
    def __init__(self, orientation="top-left", color=COLOR_ORANGE):
        super().__init__()
        self.setStyleSheet(f"""
            QFrame {{
                background-color: {color};
                border-top-left-radius: 40px;
                min-width: 120px;
                min-height: 60px;
            }}
        """)

class LcarsButton(QPushButton):
    """Pill-shaped LCARS button."""
    def __init__(self, text, color=COLOR_BLUE):
        super().__init__(text)
        self.setStyleSheet(f"""
            QPushButton {{
                background-color: {color};
                color: black;
                border-radius: 15px;
                font-weight: bold;
                font-family: 'Arial Narrow', sans-serif;
                font-size: 14px;
                padding: 5px 15px;
                text-align: right;
            }}
            QPushButton:hover {{
                background-color: white;
            }}
        """)

class LcarsWindow(QMainWindow):
    def __init__(self):
        super().__init__()
        self.setWindowTitle("LCARS Interface")
        self.resize(800, 600)
        self.setStyleSheet(f"background-color: {COLOR_BG};")

        # Main Layout Container
        central_widget = QWidget()
        self.setCentralWidget(central_widget)
        main_layout = QGridLayout(central_widget)
        main_layout.setContentsMargins(10, 10, 10, 10)
        main_layout.setSpacing(5)

        # 1. Header/Top Elbow
        self.header_elbow = LcarsElbow(color=COLOR_MAUVE)
        main_layout.addWidget(self.header_elbow, 0, 0)

        self.top_bar = QFrame()
        self.top_bar.setStyleSheet(f"background-color: {COLOR_MAUVE}; min-height: 20px;")
        main_layout.addWidget(self.top_bar, 0, 1)

        # 2. Sidebar (Navigation)
        self.sidebar = QVBoxLayout()
        self.sidebar.setSpacing(4)
        
        # Add a thick bar connecting to the elbow
        side_bar_frame = QFrame()
        side_bar_frame.setStyleSheet(f"background-color: {COLOR_ORANGE}; width: 120px;")
        
        buttons_layout = QVBoxLayout(side_bar_frame)
        buttons_layout.setAlignment(Qt.AlignTop)
        buttons_layout.addWidget(LcarsButton("NAV 01", COLOR_ORANGE))
        buttons_layout.addWidget(LcarsButton("DATA 02", COLOR_PEACH))
        buttons_layout.addWidget(LcarsButton("SENSORS", COLOR_BLUE))
        
        main_layout.addWidget(side_bar_frame, 1, 0)

        # 3. Main Content Area (Where your existing app goes)
        self.content_area = QFrame()
        self.content_area.setStyleSheet("border: 2px solid #333; border-radius: 10px;")
        content_layout = QVBoxLayout(self.content_area)
        
        dummy_label = QLabel("SYSTEM OPERATIONAL\nENTER COMMAND...")
        dummy_label.setStyleSheet("color: #9999ff; font-family: monospace; font-size: 18px;")
        dummy_label.setAlignment(Qt.AlignCenter)
        content_layout.addWidget(dummy_label)
        
        main_layout.addWidget(self.content_area, 1, 1)

        # Adjust Stretch to make content take up most space
        main_layout.setColumnStretch(1, 1)
        main_layout.setRowStretch(1, 1)

if __name__ == "__main__":
    app = QApplication(sys.argv)
    window = LcarsWindow()
    window.show()
    sys.exit(app.exec())
